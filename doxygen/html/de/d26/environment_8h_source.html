212  bool init(const Commands& commands)
213  {
214  if (commands.empty())
215  return false;
216 
217  if (commands.at(0)->getType() != CommandType::ADD_SCENE_GRAPH)
218  {
219  CONSOLE_BRIDGE_logError("When initializing environment from command history the first command must be type "
220  "ADD_SCENE_GRAPH!");
221  return false;
222  }
223 
224  auto cmd = std::static_pointer_cast<const AddSceneGraphCommand>(commands.at(0));
225  init<S>(*(cmd->getSceneGraph()));
226 
227  for (std::size_t i = 1; i < commands.size(); ++i)
228  {
229  if (!applyCommand(*(commands.at(i))))
230  {
231  CONSOLE_BRIDGE_logError("When initializing environment from command history it failed to apply a command!");
232  return false;
233  }
234  }
235 
236  init_revision_ = revision_;
237 
238  return true;
